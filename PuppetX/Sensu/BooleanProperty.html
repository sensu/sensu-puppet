def retrieve
  s = super
  case s
    when true
      :true
    when false
      :false
    else
      s
  end
end

def unsafe_munge(value)
  # downcase strings
  if value.respond_to? :downcase
    value = value.downcase
  end

  case value
    when *BooleanProperty.values_for_true
      :true
    when *BooleanProperty.values_for_false
      :false
    when :absent, 'absent'
      :absent
    else
      fail "expected a boolean value, got #{value.inspect}"
  end
end
