def insync?(is)
  ignore_labels = [
    'sensu.io/managed_by',
  ]
  # Ignore some labels that are auto generated by sensuctl
  if self.name == :labels && is.is_a?(Hash)
    new_is = {}
    is.each_pair do |key, value|
      next if ignore_labels.include?(key)
      new_is[key] = value
    end
    is = new_is
  end
  super(is)
end
