define sensu::agent::config_entry (
  Any $value,
  String[1] $key   = $name,
  String[1] $order = '50',
) {
  datacat_fragment { "sensu_agent_config-entry-${name}":
    target => 'sensu_agent_config',
    data   => {
      $key => $value,
    },
    order  => $order,
  }
}
